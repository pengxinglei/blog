原型

###原型

JS一切皆对象， 两大类型 原始值+对象
函数有区别：函数对象和普通对象
函数对象是通过new Function => typeof => function,其它的都是普通对象=> typeof => object,实例属于普通对象
function f(){} => 函数对象 ,var f1 = new f() => 普通对象
每一个函数对象都有一个prototype属性，普通对象没有
function f(){} = var f = new Function() 是简写形式
prototype下面又有constructor属性 指向当前的函数
对象都有proto属性 指向函数的原型对象

function Person(){}//空的对象
Person.prototype.name = ‘xz’
Person.prototype.age = 12
var Person1 = new Person()
var Person2 = new Person()

Person <——–| ^—>Person prototyp
prototype |–|—-constructor
_| name
| | age
| |
Person1 | | Person1
–proto– > | |—<–proto–

function cat(name,color){//普通函数
return{
name: name,
color: color
}
}
var cat1 = cat(“大明”,”黄色”)
var cat2 = cat(“小明”,”白色”)

构造函数与普通函数的区别：
都是普通函数，构造函数首字母大写，拥有this，this指向实例

function Cat(name,color){//构造函数
this.name = name,
this.color = color
this.type = “动物”
this.eat = function(){console.log(‘吃老鼠’)}
}
var cat1 = new Cat(“大明”,”黄色”)
var cat2 = new Cat(“小明”,”白色”)

//prototype(不占用内存，提高效率)
function Cat(name,color){//构造函数
this.name = name,
this.color = color
}
Cat.prototype.type = “动物”
Cat.prototype.eat = function(){console.log(‘吃老鼠’)}

var cat1 = new Cat(“大明”,”黄色”)
var cat2 = new Cat(“小明”,”白色”)

###prototype验证
in 不能判断属性方法是不是自身的还是原型上的
hasOwnPrototype 能验证属性方法是不是自身的还是原型上的
‘name’ in cat1 //true
‘type’ in cat1 //true
cat1.hasOwnPrototype(‘name’) //true
cat1.hasOwnPrototype(‘type’) //false
属性方法先在自己身上找，没有再在原型上找

###构造函数的继承
function Animal(){//动物对象
this.type = ‘动物’
}
function Cat(name,color){//猫对象
Animal.apply(this) //将父对象的构造函数绑定在子对象上
this.name = name
this.color = color
}

###apply() call() 在一个对象中调用另一个对象 apply(this,参数) 改变this指向

###区别参数不同 apply 传数组 call 一个一个传

function Animal(){//动物对象
this.type = ‘动物’
}
function Cat(name,color){//猫对象
this.name = name
this.color = color
}
Cat.prototype = new Animal()

function Animal(){}//动物对象
Animal.prototype.type = ‘动物’
function Cat(name,color){//猫对象
this.name = name
this.color = color
}
Cat.prototype = Animal.prototype
这种方式省内存，更优，缺点Cat.prototype和Animal。prototype指向了同一个对象

###apply() 与 call()的参数
function Cat(name,color){//猫对象
this.name = name
this.color = color
}
var cat = new Cat()
var o = {}
Cat.apply(o,[‘111’,’222’]) //Cat在o对象中执行
o.name //111

function Cat(name,color){//猫对象
this.name = name
this.color = color
}
var cat = new Cat()
var o = {}
Cat.call(o,’aaa’,’bbb’) //Cat在o对象中执行
o.name //aaa
