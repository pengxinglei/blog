属性检测 判断对象中是否存在属性

1、in 运算符(可以获取继承的属性)
var obj = {
name = ‘xz’,
age = 20
}
console.log(‘name’ in obj) //true => true/false

2、使用对象hasOwnProperty()方法(不可以获取继承的属性)
var obj = {
name = ‘xz’,
age = 20
}
obj.hasOwnProperty(‘name’) //true => true/false

3、!= undefined 值得判断 缺点是值不能为undefined => name = undefined
var obj = {
name = ‘xz’,
age = 20
}
console.log(obj.name !== undefined) //true => true/false

枚举属性

1、for in
var arr = [‘a’, ‘b’ ,’c’, ‘d’]
var obj = {
name = ‘xz’,
age = 20
}
for(a in obj){//for(属性 in 对象)
console.log(a) //name age
console.log(obj[a]) //xz 20 不能用obj.a => a 是变量
}
for(a in arr){//for(索引 in 数组)
console.log(a) //0 1 2 3 4
console.log(arr[a]) //a b c d
}

2、forEach() map() 都是便利数组的方法， map有返回值 ie6、7不支持
var arr = [‘a’, ‘b’ ,’c’, ‘d’]
arr.forEach(function(item,index){
console.log(item)
})

arr.map() 回调函数支持return返回值
arr.map(function(item,index){
return item
})

数组还可以用for循环
for(var i;i<10;i++){}
