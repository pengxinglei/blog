闭包与递归

##闭包
闭包就是能够读取其它函数内部变量的函数
JS中每个函数都是个体，它是可以获取父元素的信息，但是父元素不能直接获取子元素内容
关键点：是否引入了外部变量

function a(){
var n = 0;
function add(){
n++
console.log(n)
}
add()
}
a() //1

function a(){
this.n=0;
this.add = function(){
this.n++
console.log(this.n)
}
}
var a1 = new a()
a1.add()

function a(){
var n=0;
this.add = function(){
n++
console.log(n)
}
}
var a1 = new a()
a1.add()

function a(){
var n = 0;
return function add(){
n++
console.log(n)
}
add()
}
a()()

立即执行函数
var a = (function (){
var n = 0;
return function add(){
n++
console.log(n)
}
})()
a()
