this与正则表达式

this 关键字 当前

this指的调用函数的那个对象

1、函数中引用
var x =1;
function f (){
this.x;
}
console.log(x) //1

var x =1;
function f (){
this.x = 2;
}
console.log(x) //1

var x =1;
function f (){
this.x = 2;
}
f();
console.log(x) //2

//没有被绑定的对象，默认this指向window对象
var x =1;
function f (){
this.x = 2;
}
function f1 (){
this.x = 3;
}
f1();
f();
console.log(x) //2

//绑定
var f = function(){
console.log(this);//o对象
}
var o ={}
o.name = ‘xz’;
o.action = f
o.action()

var f = function(){
console.log(this);//window
}
var o ={}
o.name = ‘xz’;
o.action = f
f()
对象中引用this当前对象

构造函数引用 this
function F(name,age){
this.name = name;
this.age = age
}
var f = new F(‘ABC’,20)
f.name//ABC

function F(name,age){
this.name = name;
this.age = age
}
var f = new F(‘ABC’,20)
f.name = ‘123’
f.name//123

apply,call改变this指向，apply不传参默认为window
函数自执行如立即执行函数 this指向window

正则表达式

简单的表达式
如hi 可以匹配hi，him，等含hi的
\b 匹配一个单词的边界 匹配的独立的
\B 匹配一个单词的非边界 匹配有单词的
\d 匹配数字
\D 匹配非数字
\w 匹配字母/数字/下划线
\W 匹配除字母/数字/下划线以外的
^ 开始
$ 结束
[] 指定一堆字符中的一个
[^] 不匹配这个里面的任意一个

  • 表示前一个模式可以被重复1次或n次
  • 表示前一个模式可以被重复0次或n次
    ? 表示前一个模式可以被匹配0次或1次
    {x} 表示前一个模式可以重复X次
    {x,y} 表示前一个模式可以重复X到y次
    {x,} 表示前一个模式可以重复至少X次
    () 可以指定一堆字符来匹配一个模式 分组
    | 或者 只能匹配其中一个
